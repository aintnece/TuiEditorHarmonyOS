export function parseBlockQuote(state: ParseState): AstNode {
  const node: AstNode = new AstNode(AstNodeType.BlockQuote);
  let content: string = '';
  while (!state.isEnd()) {
    const line: string = state.currentLine();
    if (!line.startsWith('>')) break;          // ❌ 无懒续行：非 > 行直接结束
    let rest: string = line.substring(1);
    if (rest.length > 0 && rest[0] === ' ') { rest = rest.substring(1); }
    content += rest + '\n';
    state.nextLine();
  }
  const innerState: ParseState = new ParseState();
  innerState.reset(content);
  parseBlocks(innerState, node);
  return node;
}
```

## CommonMark 规则（commonmark.js 0.31.2 权威）

- **block_quote.continue**：行去掉 ≤3 前导空格后首字符是 `>` → 消费 `>` + **一个**可选空格/制表 → 继续；否则该容器不匹配。
- **lazy paragraph continuation**（`incorporateLine` 尾部）：当某容器（这里是引用块）的标记缺失、但**当前 tip 是开着的段落**且该行**非空**时，该行仍被并入那个段落（无视缺失的 `>` 标记）。即「段落可以懒续行」。**只对段落生效**（标题/分割线/围栏不懒续）。
- 嵌套 `>>> foo` = 三层引用，`> bar`/`>>baz` 行只要带前缀 `>` 就在对应层匹配、其后内容并入最内段落；完全无 `>` 的 `bar` 行靠懒续行并入最内段落。

## 目标实现（Blocks.ts；本项目「收集行→递归」模型的懒续行适配）

### 适配原理（已 Hermes 推演验证，照做）
本项目引用块是「收集 `>` 行 → 剥一层 `>` → 把内容整体递归 `parseBlocks`」。懒续行适配：
- 收集时维护 `paraOpen`（当前收集内容尾部是否有一个「开着的段落」）。
- `>` 行：剥 ≤3 空格 + `>` + 1 可选空格 → `rest`；按 `rest` 更新 `paraOpen`（`setsParaOpen`）；纯空白 `rest` 归一为 `''`。
- 非 `>` 行：若 `paraOpen && isLazyContinuable(line)` → 作为懒续行**原样**收进 content（leading 空白由内层 parseParagraph 的 trimStart 处理），`paraOpen` 保持 true；否则 **break**（引用块结束）。
- **嵌套靠递归自然解开**：`rest` 仍是 `>` 开头（如 `> foo`）时 `setsParaOpen` 乐观置 true，懒行被收进 content，递归到内层再判定——已验证 Ex250/251 正确。

### 新增 helper
```ts
/** 引用块标记剥离：跳 ≤3 前导空格 → '>' → 一个可选空格/制表。返回剥后内容；非引用行返回 null。 */
function blockQuoteMarkerStrip(line: string): string | null {
  let i: number = 0;
  let col: number = 0;
  while (i < line.length && col < 4) {
    const ch: string = line[i];
    if (ch === ' ') { col += 1; i += 1; }
    else if (ch === '\t') { col += 4 - (col % 4); i += 1; }
    else break;
  }
  if (col >= 4) return null;                       // 缩进≥4 → 非引用块
  if (i >= line.length || line[i] !== '>') return null;
  i += 1;                                           // 消费 '>'
  if (i < line.length && (line[i] === ' ' || line[i] === '\t')) { i += 1; }   // 一个可选空格/制表
  return line.substring(i);
}

/** 引用块起始（容忍 ≤3 前导空格）。 */
function isBlockQuoteStart(line: string): boolean {
  return blockQuoteMarkerStrip(line) !== null;
}

/** 收进 rest 后，引用块尾部是否有「开着的段落」（决定后续非 > 行能否懒续）。
 *  blank / ATX 标题(#) / 围栏(```/~~~) / 分割线 → false（无开段落或不可懒续）；
 *  其余（普通段落文本、嵌套 > 行、列表项等）→ true（乐观；递归会正确收口）。 */
function setsParaOpen(rest: string): boolean {
  if (isBlankLine(rest)) return false;
  if (rest.startsWith('#')) return false;
  if (rest.startsWith('```') || rest.startsWith('~~~')) return false;
  if (isThematicBreak(rest)) return false;
  return true;
}

/** 非 > 行能否作为段落懒续行（= parseParagraph 不会被它打断）。
 *  缩进≥4 的行本批保守不收（Ex238 缩进列表懒续行留后续，避免动 parseParagraph 列表打断规则）。 */
function isLazyContinuable(line: string): boolean {
  if (isBlankLine(line)) return false;
  if (countIndentColumns(line) >= 4) return false;     // 本批不处理缩进懒续行（Ex238 延后）
  if (line.startsWith('#')) return false;              // ATX 标题打断段落
  if (line.startsWith('```') || line.startsWith('~~~')) return false;  // 围栏打断
  if (isThematicBreak(line)) return false;             // 分割线打断
  if (line.startsWith('>')) return false;              // > 行另行处理
  if (isHtmlBlockInterrupt(line)) return false;        // HTML 块 type1-6 打断
  if (isBulletListMarker(line) || isOrderedListMarker(line)) return false;  // 非缩进列表标记打断
  return true;
}
```

### 重写 `parseBlockQuote`
```ts
export function parseBlockQuote(state: ParseState): AstNode {
  const node: AstNode = new AstNode(AstNodeType.BlockQuote);
  let content: string = '';
  let paraOpen: boolean = false;
  while (!state.isEnd()) {
    const line: string = state.currentLine();
    const rest: string | null = blockQuoteMarkerStrip(line);
    if (rest !== null) {
      // > 行：纯空白 rest 归一为空行（Ex241）
      const piece: string = isBlankLine(rest) ? '' : rest;
      content += piece + '\n';
      paraOpen = setsParaOpen(rest);
      state.nextLine();
    } else {
      // 非 > 行：懒续行候选
      if (paraOpen && isLazyContinuable(line)) {
        content += line + '\n';     // 原样收（内层 parseParagraph 会 trimStart）
        // paraOpen 保持 true
        state.nextLine();
      } else {
        break;                       // 引用块结束
      }
    }
  }
  const innerState: ParseState = new ParseState();
  innerState.reset(content);
  parseBlocks(innerState, node);
  return node;
}
```

## 逐例验证逻辑（实现后应满足）
- **230** `   > # Foo\n   > bar\n > baz`：每行 strip ≤3 空格 + `>` + 空格 → `# Foo`/`bar`/`baz`；`# Foo` 后 paraOpen=false，`bar` 后 true，`baz` 是 `>` 行续。content=`# Foo\nbar\nbaz` → `<blockquote><h1>Foo</h1><p>bar\nbaz</p></blockquote>`。
- **232** `> # Foo\n> bar\nbaz`：`# Foo`(paraOpen=false) / `bar`(paraOpen=true) / `baz` 非 > 且 paraOpen → 懒续 → content `# Foo\nbar\nbaz` → 同上。
- **233** `> bar\nbaz\n> foo`：`bar`(true) / `baz` 懒续(true) / `> foo` > 行 → content `bar\nbaz\nfoo` → `<blockquote><p>bar\nbaz\nfoo</p></blockquote>`。
- **247** `> bar\nbaz`：`bar`(true) / `baz` 懒续 → `<blockquote><p>bar\nbaz</p></blockquote>`。
- **241** `>\n> foo\n>  `：`>`→`''`(blank, paraOpen=false) / `> foo`→`foo`(true) / `>  `→strip `>`+1空格=` `→ isBlankLine→归一 `''`(paraOpen=false)。content=`\nfoo\n\n` → `<blockquote><p>foo</p></blockquote>`。
- **250** `> > > foo\nbar`：`> > foo`(setsParaOpen 乐观 true，因非 #/围栏/分割线/blank) / `bar` 懒续。content=`> > foo\nbar` → 递归：内层 `> foo`+`bar` → 再内层 `foo`+`bar` → `<blockquote><blockquote><blockquote><p>foo\nbar</p></blockquote></blockquote></blockquote>`。
- **251** `>>> foo\n> bar\n>>baz`：`>>> foo`→剥1层`>`(无空格)→`>> foo`(paraOpen 乐观 true) / `> bar`→`bar`(> 行) / `>>baz`→剥1层→`>baz`(> 行)。content=`>> foo\nbar\n>baz` → 递归逐层解开 → 三层引用、最内 `<p>foo\nbar\nbaz</p>`。
- **238**（不修）`> foo\n    - bar`：`    - bar` 缩进4 → isLazyContinuable 返回 false → 引用块在 foo 后结束 → **保持现状（仍 FAIL），不得回归**。
